drop table policys;
drop table property;
drop table quote;
drop table HomeownerInfo;
drop table Location;
drop table Login;

create table Login(
USER_NAME varchar2
USER_PWD varchar2
ROLE varchar2(255) NOT NULL,
)

create table Location(
QUOTE_ID NUMBER(10),
RESIDENCE_TYPE VARCHAR2(255)
ADDRESS_LINE_1 NVARCHAR2(500),
ADDRESS_LINE_2 NVARCHAR2(500),
CITY VARCHAR2(255),
STATE VARCHAR2(255),
ZIP VARCHAR2(255),
RESIDENCE_USE VARCHAR2(255),
USER_NAME VARCHAR2(255),
CONSTRAINT RESICENTTYPE_ck CHECK(RESIDENCE_TYPE IN ('Single-Family Home', 'Condo', 'Townhouse', 'Rowhouse', 'Duplex', 'Apartment')),
CONSTRAINT RESIDENCE_USE_ck CHECK( RESIDENCE_USE IN ('Primary', 'Secondary', 'Rental Property'))
);

CREATE TABLE HomeownerInfo(
QUOTE_ID NUMBER(10),
FIRST_NAME VARCHAR2(255),
LAST_NAME VARCHAR2(255),
DOB VARCHAR2(255),
IS_RETIRED VARCHAR2(255),
SSN VARCHAR2(255),
EMAIL_ADDRESS VARCHAR2(255)
);

CREATE TABLE Quote(
QUOTE_ID NUMBER(10),
PREMIUM NUMBER(10, 2),
DWELLING_COVERAGE NUMBER(10, 2),
DETACHED_STRUCTURE NUMBER(10, 2),
PERSONAL_PROPERTY NUMBER(10, 2),
ADDNL_LIVING_EXPENSE NUMBER(10, 2),
MEDICAL_EXPENSE NUMBER(10, 2),
DEDUCTIBLE NUMBER(10, 2)
);

CREATE TABLE Property(
QUOTE_ID NUMBER(10),
MARKET_VALUE NUMBER(10),
YEAR_BUILT NUMBER(10),
SQUARE_FOOTAGE NUMBER(10),
DWELLING_STYLE NUMBER(10),
ROOF_MATERIAL VARCHAR2(255),
GARAGE_TYPE VARCHAR2(255),
NUM_FULL_BATHS NUMBER(10),
NUM_HALF_BATHS NUMBER(10),
HAS_SWIMMING_POOL VARCHAR2(255) DEFAULT 'FALSE' NOT NULL 
);

CREATE TABLE Policys(
POLICY_KEY VARCHAR2(255),
QUOTE_ID NUMBER(10),
POLICY_EFFECTIVE_DATE VARCHAR2(255),
POLICY_END_DATE VARCHAR2(255),
POLICY_TERM VARCHAR2(255),
POLICY_STATUS VARCHAR2(255),
CONSTRAINT POLICY_DATE_CK CHECK( POLICY_EFFECTIVE_DATE < POLICY_END_DATE )
);

commit;
